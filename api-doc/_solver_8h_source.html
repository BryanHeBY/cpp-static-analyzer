149 template <typename Fact>
150 class WorkListSolver: public AbstractSolver<Fact> {
151 protected:
152
153 void doSolveForward(
154 const std::unique_ptr<DataflowAnalysis<Fact>>& dataflowAnalysis,
155 std::shared_ptr<fact::DataflowResult<Fact>> result) const override
156 {
157 std::queue<std::shared_ptr<ir::Stmt>> workList;
158 std::shared_ptr<graph::CFG> cfg = dataflowAnalysis->getCFG();
159 workList.push(cfg->getEntry());
160 for (std::shared_ptr<ir::Stmt>& s : cfg->getIR()->getStmts()) {
161 workList.push(s);
162 }
163 workList.push(cfg->getExit());
164 while (!workList.empty()) {
165 std::shared_ptr<ir::Stmt> stmt = workList.front();
166 workList.pop();
167 if (stmt == cfg->getEntry()) {
168 continue;
169 }
170 for (const std::shared_ptr<ir::Stmt>& pred : cfg->getPredsOf(stmt)) {
171 dataflowAnalysis->meetInto(result->getOutFact(pred),
172 result->getInFact(stmt));
173 }
174 if (dataflowAnalysis->transferNode(stmt,
175 result->getInFact(stmt), result->getOutFact(stmt))) {
176 for (const std::shared_ptr<ir::Stmt>& succ : cfg->getSuccsOf(stmt)) {
177 workList.push(succ);
178 }
179 }
180 }
181 }
